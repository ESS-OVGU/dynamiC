403 float fast_approx_pow (float x, float p)
404 {
405  union { float f;
406  unsigned int i;
407  } vx = { x };
408 
409  union { unsigned int i;
410  float f;
411  } mx = { (vx.i & 0x007FFFFF) | 0x3f000000 };
412 
413  x = vx.i * 1.1920928955078125e-7f
414  - 124.22551499f
415  - 1.498030302f * mx.f
416  - 1.72587999f / (0.3520887068f + mx.f);
417 
418  x *= p;
419 
420  float offset = (x < 0) ? 1.0f : 0.0f;
421  float clipp = (x < -126) ? -126.0f : x;
422  float z = clipp - (int) clipp + offset;
423  mx.i = (unsigned int) ( (1 << 23) * (clipp + 121.2740575f + 27.7280233f / (4.84252568f - z) - 1.49012907f * z) );
424

669 dyn_char dyn_op_cmp (dyn_c* dyn1, dyn_c* dyn2)
670 {
671  enum{EQ,LT,GT,NEQ,TYPE,MARK=0xff};//0,1,2,3,4
672 
673  dyn_c *tmp = DYN_IS_REFERENCE(dyn1) ? dyn1->data.ref : dyn1;
674  dyn_ushort ret;
675  dyn_c tmp2;
676  DYN_INIT(&tmp2);
677  dyn_ushort i;
678 
679  if(DYN_IS_REFERENCE(dyn2))
680  dyn2=dyn2->data.ref;
681 
682  if (DYN_IS_NONE(tmp) && DYN_IS_NONE(dyn2))
683  goto GOTO_EQ;
684 
685  if (DYN_IS_NONE(tmp) && DYN_NOT_NONE(dyn2))
686  goto GOTO_LT;
687 
688  if (DYN_NOT_NONE(tmp) && DYN_IS_NONE(dyn2))
689  goto GOTO_GT;
690 
691  //this switch has no breaks goto is used to leave to desired return
692  switch (max_type(tmp, dyn2)) {
693  case BOOL:
694  if (dyn_get_bool(tmp) < dyn_get_bool(dyn2))
695  goto GOTO_LT;
696  if (dyn_get_bool(tmp) > dyn_get_bool(dyn2))
697  goto GOTO_GT;
698  goto GOTO_EQ;
699  case INTEGER:
700  if (dyn_get_int(tmp) < dyn_get_int(dyn2))
701  goto GOTO_LT;
702  if (dyn_get_int(tmp) > dyn_get_int(dyn2))
703  goto GOTO_GT;
704  goto GOTO_EQ;
705  case FLOAT:
706  if (dyn_get_float(tmp) < dyn_get_float(dyn2))
707  goto GOTO_LT;
708  if (dyn_get_float(tmp) > dyn_get_float(dyn2))
709  goto GOTO_GT;
710  goto GOTO_EQ;
711  case STRING: {
712  if (DYN_TYPE(tmp) != DYN_TYPE(dyn2))
713  goto GOTO_TYPE;
714  i = dyn_strcmp(tmp->data.str, dyn2->data.str);
715  if (i < 0)
716  goto GOTO_LT;
717  if (i > 0)
718  goto GOTO_GT;
719  goto GOTO_EQ;
720  }
721  case SET: {
722  if (DYN_TYPE(tmp) != DYN_TYPE(dyn2))
723  goto GOTO_TYPE;
724  {
725  dyn_c * lset;
726  dyn_c * rset;
727  if (DYN_LIST_LEN(tmp) == DYN_LIST_LEN(dyn2)){
728  // list might be EQ
729  ret = EQ;
730  lset = tmp;
731  rset = dyn2;
732  } else if (DYN_LIST_LEN(tmp) < DYN_LIST_LEN(dyn2)){
733  // list might be EQ
734  ret = LT;
735  lset = tmp;
736  rset = dyn2;
737  } else if (DYN_LIST_LEN(tmp) > DYN_LIST_LEN(dyn2)){
738  // list might be GT
739  ret = GT;
740  lset = dyn2;
741  rset = tmp;
742  }
743  //check if all elements of lset are in rset
744  //break to NEQ or complete to confirm
745  for (i=0; i<DYN_LIST_LEN(lset); ++i) {
746  dyn_set_ref(&tmp2, DYN_LIST_GET_REF(lset, i));
747  dyn_op_in(&tmp2, rset);
748  if (!dyn_get_bool(&tmp2)){
749  ret = NEQ;
750  break;
751  }
752 
753  }
754  goto GOTO_RET;
755 
756  }
757  //unreachable
758  }
759  case LIST: {
760  if (DYN_TYPE(tmp) != DYN_TYPE(dyn2))
761  goto GOTO_TYPE;
762  if (DYN_LIST_LEN(tmp) < DYN_LIST_LEN(dyn2))
763  goto GOTO_LT;
764  if (DYN_LIST_LEN(tmp) > DYN_LIST_LEN(dyn2))
765  goto GOTO_GT;
766  if (DYN_LIST_LEN(tmp) == DYN_LIST_LEN(dyn2)) {
767  if (DYN_LIST_LEN(tmp) == 0)
768  goto GOTO_EQ;
769 
770  //put marker into return value
771  ret = MARK;
772  for (i=0; i<DYN_LIST_LEN(tmp); ++i) {
773  dyn_set_ref(&tmp2, DYN_LIST_GET_REF(tmp, i));
774  ret = dyn_op_cmp(&tmp2, DYN_LIST_GET_REF(dyn2, i));
775  if (ret != EQ)
776  break;
777  }
778  //reached the end of the for loop
779  goto GOTO_RET;
780  }
781  //unreachable
782  }
783  }
784  //unreachable
785  // break with diffent type (string, list and set)
786  goto GOTO_TYPE;
787 GOTO_TYPE:
788  ret = TYPE;
789  goto GOTO_RET;
790 
791 GOTO_EQ:
792  ret = EQ;
793  goto GOTO_RET;
794 GOTO_LT:
795  ret = LT;
796  goto GOTO_RET;
797 GOTO_GT:
798  ret = GT;
799  goto GOTO_RET;
800 /*
801 GOTO_NEQ:
802  ret = NEQ;
803  goto GOTO_RET;
804 */
805 GOTO_RET:
806  //dyn_set_int(dyn1, ret);
807  return ret;
808 
809 }
